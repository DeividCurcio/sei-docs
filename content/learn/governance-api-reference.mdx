  ### Advanced Query Patterns

  ```bash copy
  # Filter proposals by status
  seid q gov proposals --status voting_period --output json

  # Get proposals with pagination
  seid q gov proposals --limit 10 --offset 20

  # Filter by depositor
  seid q gov proposals --depositor $(seid keys show mykey -a)

  # Get expedited proposals only
  seid q gov proposals --output json | jq '.proposals[] | select(.is_expedited == true)'

  # Monitor proposal progression
  watch -n 30 'seid q gov tally 1'
  ```

  ```json copy
  }
  ```

  ## Event Monitoring

  ### WebSocket Subscriptions

  ```javascript copy
  // Connect to Sei WebSocket
  const ws = new WebSocket('wss://rpc.sei-apis.com/websocket');

  // Subscribe to proposal events
  const subscriptions = {
    proposals: {
      jsonrpc: "2.0",
      method: "subscribe",
      id: 1,
      params: {
        query: "tm.event='Tx' AND submit_proposal.proposal_id EXISTS"
      }
    },
    votes: {
      jsonrpc: "2.0", 
      method: "subscribe",
      id: 2,
      params: {
        query: "tm.event='Tx' AND proposal_vote.proposal_id EXISTS"
      }
    }
  };

  // Handle events
  ws.onmessage = (event) => {
    const data = JSON.parse(event.data);
    
    if (data.result?.data?.type === 'tendermint/event/Tx') {
      const events = data.result.data.value.TxResult.result.events;
      
      events.forEach(evt => {
        switch(evt.type) {
          case 'submit_proposal':
            handleNewProposal(evt.attributes);
            break;
          case 'proposal_vote':
            handleVote(evt.attributes);
            break;
        }
      });
    }
  };
  ```

  ## EVM Integration

  ```solidity copy
  // SPDX-License-Identifier: MIT
  pragma solidity ^0.8.0;

  contract GovernanceParticipant {
      event ProposalSubmitted(uint64 indexed proposalId, address indexed submitter);
      event VoteCast(uint64 indexed proposalId, address indexed voter, int32 option);
      
      IGov constant GOV = IGov(0x0000000000000000000000000000000000001006);
      
      function submitAndTrack(string calldata proposalJSON) external payable returns (uint64) {
          uint64 proposalId = GOV.submitProposal{value: msg.value}(proposalJSON);
          emit ProposalSubmitted(proposalId, msg.sender);
          return proposalId;
      }
      
      function voteAndTrack(uint64 proposalId, int32 option) external {
          require(GOV.vote(proposalId, option), "Vote failed");
          emit VoteCast(proposalId, msg.sender, option);
      }
  }
  ```

  ## Production Integration Example

  ```typescript copy
  class GovernanceMonitor {
    private client: SeiQueryClient;
    private cache = new Map<string, { data: any; timestamp: number }>();
    private cacheTTL = 30000; // 30 seconds

    constructor(restEndpoint: string, wsEndpoint: string) {
      this.client = new SeiQueryClient(restEndpoint);
      this.setupWebSocket(wsEndpoint);
    }

    async getProposal(id: string) {
      const cached = this.cache.get('proposal-' + id);
      if (cached && Date.now() - cached.timestamp < this.cacheTTL) {
        return cached.data;
      }

      const proposal = await this.client.gov.proposal(id);
      this.cache.set('proposal-' + id, { 
        data: proposal, 
        timestamp: Date.now() 
      });
      
      return proposal;
    }

    async calculateParticipationRate(proposalId: string) {
      const tally = await this.client.gov.tally(proposalId);
      const pool = await this.client.staking.pool();
      
      const totalVotes = BigInt(tally.yes) + BigInt(tally.no) + 
                        BigInt(tally.abstain) + BigInt(tally.noWithVeto);
      
      return Number(totalVotes) / Number(pool.bondedTokens);
    }
  }
  ```
